    Program Python Deret MacLaurin

    Program untuk persamaan deferensial $$e^{x-\cos \left( x\right) }$$ sampai error yang dihasilkan kurang dari 0.001 untuk x=1

    import numpy as np

x = 1
e = 0

def f(x):
    return e**x - np.cos(x)

a = np.zeros(40)
b = np.zeros(40)
c = np.zeros(40)

a[0] = 1
b[0] = 8

c[0] = (a[0] + b[0])/2
for n in range(9):
    if f(a[n])*f(c[n])<0:
        a[n+1] = a[n]
        b[n+1] = c[n]
    else:
        a[n+1] = c[n]
        b[n+1] = b[n]
    c[n+1] = (a[n+1] + b[n+1])/2
    print(c[n+1],f(c[n+1]))
